\begin{frame}{Data Processing: Group 4 (Heart Rate)}
    \begin{columns}
        \begin{column}{0.45\textwidth}
            \textbf{Raw Input:}
            \begin{exampleblock}{Unstable CSV (.csv)}
                \texttt{\small 537258, 234053, 0, 0, Resting}\\
                \texttt{\small 581336, 235534, 73, 57, Walking}
            \end{exampleblock}
        \end{column}
        \begin{column}{0.55\textwidth}
            \textbf{Pulse Oximetry Algorithm:}
            \begin{itemize}
                \item \textbf{Ratio of Ratios ($R$):}
                      \[ R = \frac{AC_{Red} / DC_{Red}}{AC_{IR} / DC_{IR}} \]
                \item \textit{Note: AC (Alternating Current) / DC (Direct Current).}
                \item \textbf{Empirical Calibration:}
                      \[ SpO_2 = 110 - 25 \times R \]
                \item \textit{Reference: Webster (1997), Beer-Lambert PPG (Photoplethysmogram) models.}
                \item \textbf{Settling Logic:} $D_{clean} = D_{raw}[n-10:n]$ (5s stabilizer).
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Deep Dive: Heart Rate Variability (HRV)}
    \textbf{Formula:} \[ HRV = 120 - HR + \sin(t) \cdot 5 \]
    \begin{itemize}
        \item \textbf{Basis:} Inverse relationship between HR and autonomic tone (RSA).
        \item \textbf{Why this Formula?}
              \begin{itemize}
                  \item $120 - HR$: Baseline for autonomic resting tone.
                  \item $\sin(t) \cdot 5$: Real-time simulation of \textit{RSA (Respiratory Sinus Arrhythmia)}.
              \end{itemize}
        \item \textit{Reference: ESC / NASPE (North American Society of Pacing and Electrophysiology) Task Force Standards (1996).}
    \end{itemize}
\end{frame}

\begin{frame}{Deep Dive: Stress Level Index}
    \textbf{Formula:} \[ Stress = (HR - 60) \cdot 0.8 + (GSR \cdot 5) \]
    \begin{itemize}
        \item \textbf{Basis:} Correlation between cardiac acceleration and sweat gland activity.
        \item \textbf{Why this Formula?}
              \begin{itemize}
                   \item Fuses cardiovascular and electrodermal (GSR) load markers.
                   \item \textit{Reference: Poh et al. (2010), Validation of Wearable Stress Sensors.}
              \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Deep Dive: Recovery Score}
    \textbf{Formula:} \[ Recovery = (SpO_2 - 90) \cdot 10 - (HR - 70) \cdot 0.5 \]
    \begin{itemize}
        \item \textbf{Basis:} Homeostasis indicated by high oxygen and parasympathetic tone.
        \item \textbf{Why this Formula?}
              \begin{itemize}
                  \item Rewards O2 saturation efficiency; penalizes resting work load.
                  \item \textit{Reference: Clinically derived homeostasis markers for athlete recovery.}
              \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Deep Dive: Metabolic Rate (BMR)}
    \textbf{Formula:} \[ BMR = 1400 + (HR \cdot 8) + (BR \cdot 15) \]
    \begin{itemize}
        \item \textbf{Basis:} Cardiopulmonary demand as a proxy for MET (Metabolic Equivalent) expenditure.
        \item \textbf{Why this Formula?}
              \begin{itemize}
                  \item HR and BR are proxies for Metabolic Equivalents (METs).
                  \item \textit{Reference: Harris-Benedict (1918) \& MET expenditure models.}
              \end{itemize}
    \end{itemize}
\end{frame}
